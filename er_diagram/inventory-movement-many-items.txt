table organizations{
  id bigint
  name varchar(100)
}

table departments{
  id bigint
  name varchar(100)
  address varchar(200)
  organization_id bigint
}

table items{
  category_id bigint
  name varchar(100)
  description varchar(1000)
  serviceable boolean
  id bigint
  inv_number varchar(50)
  created timestamp
  finished timestamp
}

table categories{
  id bigint
  name varchar(100)
  organization_id bigint
}

table users{
  id bigint
  name varchar(100)
  reporting boolean
  position varchar(50)
  email varchar(50)
  password varchar(50)
  role varchar(10)
  organization_id bigint
}

table movements{
  id bigint
  movement_date timestamp
  from_owner_id bigint
  to_owner_id bigint
  from_department_id bigint
  to_department_id bigint
  description varchar(1000)
}

table movements_items{
  movement_id bigint
  item_id bigint
}

ref: categories.id < items.category_id
ref: organizations.id < categories.organization_id
ref: organizations.id < departments.organization_id
ref: organizations.id < users.organization_id
ref: departments.id < movements.from_department_id
ref: departments.id < movements.to_department_id
ref: users.id < movements.from_owner_id
ref: users.id < movements.to_owner_id
ref: items.id < movements_items.item_id
ref: movements.id < movements_items.movement_id
