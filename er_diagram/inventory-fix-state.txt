table organizations{
  id bigint
  name varchar(100)
}

table departments{
  id bigint
  name varchar(100)
  address varchar(200)
  organization_id bigint
}

table items{
  category_id bigint
  name varchar(100)
  description varchar(1000)
  serviceable boolean
  id bigint
  inv_number varchar(50)
  start timestamp
  end timestamp
  owner_id bigint
  user_id bigint
  department_id bigint
}

table categories{
  id bigint
  name varchar(100)
  organization_id bigint
}

table users{
  id bigint
  name varchar(100)
  reporting boolean
  position varchar(50)
  email varchar(50)
  password varchar(50)
  role varchar(10)
  organization_id bigint
}

ref: categories.id < items.category_id
ref: organizations.id < categories.organization_id
ref: organizations.id < departments.organization_id
ref: organizations.id < users.organization_id
ref: users.id < items.owner_id
ref: users.id < items.user_id
ref: departments.id < items.department_id
